### 가장 긴 바이토닉 수열
https://www.acmicpc.net/problem/11054
```python
import sys

input = sys.stdin.readline

n = int(input().strip())
seq = list(map(int, input().split()))

# i 증가 수열
left = [1] * n
for i in range(n):
    best = 1
    for j in range(i):
        if seq[j] < seq[i]:
            if left[j] + 1 > best:
                best = left[j] + 1
    left[i] = best

# i 감소 수열
right = [1] * n
for i in range(n - 1, -1, -1):
    best = 1
    for j in range(i + 1, n):
        if seq[j] < seq[i]:
            if right[j] + 1 > best:
                best = right[j] + 1
    right[i] = best

# i가 max인 바이토닉 길이
answer = 0
for i in range(n):
    peak_len = left[i] + right[i] - 1
    if peak_len > answer:
        answer = peak_len

print(answer)
```
